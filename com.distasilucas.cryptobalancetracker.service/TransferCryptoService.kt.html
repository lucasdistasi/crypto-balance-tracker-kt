package com.distasilucas.cryptobalancetracker.service

import com.distasilucas.cryptobalancetracker.model.request.crypto.TransferCryptoRequest
import com.distasilucas.cryptobalancetracker.model.response.crypto.FromPlatform
import com.distasilucas.cryptobalancetracker.model.response.crypto.ToPlatform
import com.distasilucas.cryptobalancetracker.model.response.crypto.TransferCryptoResponse
import io.github.oshai.kotlinlogging.KotlinLogging
import org.springframework.stereotype.Service
import java.math.BigDecimal

@Service
class TransferCryptoService(
    private val userCryptoService: UserCryptoService,
    private val platformService: PlatformService
) {

    private val logger = KotlinLogging.logger { }

    fun transferCrypto(transferCryptoRequest: TransferCryptoRequest): TransferCryptoResponse {
        val toPlatformResponse = platformService.retrievePlatformById(transferCryptoRequest.toPlatform!!)
        val userCryptoToTransfer = userCryptoService.findByUserCryptoId(transferCryptoRequest.userCryptoId!!)
        val fromPlatformResponse = platformService.retrievePlatformById(userCryptoToTransfer.platformId)

        if (isToAndFromSame(toPlatformResponse.id, userCryptoToTransfer.id)) {
            throw RuntimeException("From platform and to platform cannot be the same")
        }

        val actualCryptoQuantity = userCryptoToTransfer.quantity
        val networkFee = transferCryptoRequest.networkFee!!
        val quantityToTransfer = transferCryptoRequest.quantityToTransfer!!
        val totalToSubtract = calculateTotalToSubtract(actualCryptoQuantity, transferCryptoRequest)

        if (totalToSubtract > actualCryptoQuantity) {
            throw InsufficientBalanceException("You don't have enough balance to perform this action")
        }

        val toPlatformOptionalUserCrypto = userCryptoService.findByCoingeckoCryptoIdAndPlatformId(
            userCryptoToTransfer.coingeckoCryptoId,
            transferCryptoRequest.toPlatform
        )
        val remainingCryptoQuantity = calculateRemainingCryptoQuantity(actualCryptoQuantity, transferCryptoRequest)

        var transferCryptoResponse: TransferCryptoResponse? = null

        if (doesFromPlatformHaveRemaining(remainingCryptoQuantity) && toPlatformOptionalUserCrypto.isPresent) {
            val quantityToSendReceive = calculateQuantityToSendReceive(transferCryptoRequest)
            val toPlatformUserCrypto = toPlatformOptionalUserCrypto.get()
            val newQuantity = toPlatformUserCrypto.quantity.add(quantityToSendReceive)

            val updatedFromPlatformUserCrypto = userCryptoToTransfer.copy(quantity = remainingCryptoQuantity)
            val updatedToPlatformUserCrypto = toPlatformUserCrypto.copy(quantity = newQuantity)

            // userCryptoService.saveOrUpdateAll(listOf(updatedFromPlatformUserCrypto, updatedToPlatformUserCrypto))

            val fromPlatform = FromPlatform(
                userCryptoId = transferCryptoRequest.userCryptoId,
                networkFee = networkFee,
                quantityToTransfer = quantityToTransfer,
                totalToSubtract = totalToSubtract,
                quantityToSendReceive = quantityToSendReceive,
                remainingCryptoQuantity = remainingCryptoQuantity,
                sendFullQuantity = transferCryptoRequest.sendFullQuantity!!
            )
            val toPlatform = ToPlatform(
                platformId = transferCryptoRequest.toPlatform,
                newQuantity = newQuantity
            )
            transferCryptoResponse = TransferCryptoResponse(fromPlatform, toPlatform)
        }

        if (doesFromPlatformHaveRemaining(remainingCryptoQuantity) && toPlatformOptionalUserCrypto.isEmpty) {
            val quantityToSendReceive = calculateQuantityToSendReceive(transferCryptoRequest)
            val toPlatformUserCrypto = userCryptoToTransfer.copy(
                quantity = quantityToSendReceive,
                platformId = transferCryptoRequest.toPlatform
            )
            val updatedUserCryptoToTransfer = userCryptoToTransfer.copy(quantity = remainingCryptoQuantity)

            // userCryptoService.saveOrUpdateAll(listOf(toPlatformUserCrypto, updatedUserCryptoToTransfer))

            val fromPlatform = FromPlatform(
                userCryptoId = transferCryptoRequest.userCryptoId,
                networkFee = networkFee,
                quantityToTransfer = quantityToTransfer,
                totalToSubtract = totalToSubtract,
                quantityToSendReceive = quantityToSendReceive,
                remainingCryptoQuantity = remainingCryptoQuantity,
                sendFullQuantity = transferCryptoRequest.sendFullQuantity!!
            )
            val toPlatform = ToPlatform(
                platformId = transferCryptoRequest.toPlatform,
                newQuantity = quantityToSendReceive
            )
            transferCryptoResponse = TransferCryptoResponse(fromPlatform, toPlatform)
        }

        if (!doesFromPlatformHaveRemaining(remainingCryptoQuantity) && toPlatformOptionalUserCrypto.isPresent) {
            val toPlatformUserCrypto = toPlatformOptionalUserCrypto.get()
            val quantityToSendReceive = quantityToTransfer.minus(networkFee)
            val newQuantity = toPlatformUserCrypto.quantity.add(quantityToSendReceive)

            val updatedToPlatformUserCrypto = toPlatformUserCrypto.copy(quantity = newQuantity)

            // userCryptoService.deleteUserCrypto(userCryptoToTransfer.id)
            // userCryptoService.saveOrUpdateAll(listOf(updatedToPlatformUserCrypto))

            val fromPlatform = FromPlatform(
                userCryptoId = transferCryptoRequest.userCryptoId,
                networkFee = networkFee,
                quantityToTransfer = quantityToTransfer,
                totalToSubtract = totalToSubtract,
                quantityToSendReceive = quantityToSendReceive,
                remainingCryptoQuantity = remainingCryptoQuantity,
                sendFullQuantity = transferCryptoRequest.sendFullQuantity!!
            )
            val toPlatform = ToPlatform(
                platformId = transferCryptoRequest.toPlatform,
                newQuantity = newQuantity
            )
            transferCryptoResponse = TransferCryptoResponse(fromPlatform, toPlatform)
        }

        if (!doesFromPlatformHaveRemaining(remainingCryptoQuantity) && toPlatformOptionalUserCrypto.isEmpty) {
            val quantityToSendReceive = quantityToTransfer.minus(networkFee)
            val updatedFromPlatformUserCrypto = userCryptoToTransfer.copy(
                quantity = quantityToSendReceive,
                platformId = toPlatformResponse.id
            )

            // userCryptoService.saveOrUpdateAll(listOf(updatedFromPlatformUserCrypto))

            val fromPlatform = FromPlatform(
                userCryptoId = transferCryptoRequest.userCryptoId,
                networkFee = networkFee,
                quantityToTransfer = quantityToTransfer,
                totalToSubtract = totalToSubtract,
                quantityToSendReceive = quantityToSendReceive,
                remainingCryptoQuantity = remainingCryptoQuantity,
                sendFullQuantity = transferCryptoRequest.sendFullQuantity!!
            )
            val toPlatform = ToPlatform(
                platformId = transferCryptoRequest.toPlatform,
                newQuantity = quantityToSendReceive
            )
            transferCryptoResponse = TransferCryptoResponse(fromPlatform, toPlatform)
        }

        logger.info { "Transferred $quantityToTransfer ${userCryptoToTransfer.coingeckoCryptoId} from platform ${fromPlatformResponse.name} to ${toPlatformResponse.name}" }

        return transferCryptoResponse!!
    }

    private fun isToAndFromSame(toPlatformId: String, fromPlatformId: String) = toPlatformId == fromPlatformId

    private fun calculateTotalToSubtract(
        actualCryptoQuantity: BigDecimal,
        transferCryptoRequest: TransferCryptoRequest
    ): BigDecimal {
        val quantityToTransfer = transferCryptoRequest.quantityToTransfer!!
        val networkFee = transferCryptoRequest.networkFee!!
        val totalToSubtract = quantityToTransfer.add(networkFee)

        return if (transferCryptoRequest.sendFullQuantity == true) {
            if (actualCryptoQuantity >= totalToSubtract) totalToSubtract else actualCryptoQuantity
        } else {
            return quantityToTransfer
        }
    }

    private fun calculateQuantityToSendReceive(transferCryptoRequest: TransferCryptoRequest): BigDecimal {
        val quantityToTransfer = transferCryptoRequest.quantityToTransfer!!
        val networkFee = transferCryptoRequest.networkFee!!

        return if (transferCryptoRequest.sendFullQuantity == true) quantityToTransfer else quantityToTransfer.subtract(
            networkFee
        )
    }

    private fun calculateRemainingCryptoQuantity(
        actualCryptoQuantity: BigDecimal,
        transferCryptoRequest: TransferCryptoRequest
    ): BigDecimal {
        return if (transferCryptoRequest.sendFullQuantity == true) {
            val totalToSubtract = calculateTotalToSubtract(actualCryptoQuantity, transferCryptoRequest)
            calculateRemainingCryptoQuantity(actualCryptoQuantity, totalToSubtract)
        } else {
            calculateRemainingCryptoQuantity(actualCryptoQuantity, transferCryptoRequest.quantityToTransfer!!)
        }
    }

    private fun calculateRemainingCryptoQuantity(
        actualCryptoQuantity: BigDecimal,
        totalToSubtract: BigDecimal
    ) = if (actualCryptoQuantity > totalToSubtract) actualCryptoQuantity.subtract(totalToSubtract) else BigDecimal.ZERO

    private fun doesFromPlatformHaveRemaining(remainingCryptoQuantity: BigDecimal) =
        remainingCryptoQuantity > BigDecimal.ZERO
}
