package com.distasilucas.cryptobalancetracker.service

import com.distasilucas.cryptobalancetracker.entity.Crypto
import com.distasilucas.cryptobalancetracker.model.response.insights.Balances
import com.distasilucas.cryptobalancetracker.model.response.insights.CryptoInsights
import com.distasilucas.cryptobalancetracker.model.response.insights.platform.PlatformInsightsResponse
import io.github.oshai.kotlinlogging.KotlinLogging
import org.springframework.stereotype.Service
import java.math.BigDecimal
import java.math.RoundingMode

@Service
class InsightsService(
    private val platformService: PlatformService,
    private val userCryptoService: UserCryptoService,
    private val cryptoService: CryptoService
) {

    private val logger = KotlinLogging.logger { }

    fun retrievePlatformInsights(platformId: String): PlatformInsightsResponse {
        logger.info { "Retrieving insights for platform with id $platformId" }

        val platformResponse = platformService.retrievePlatformById(platformId)
        val userCryptos = userCryptoService.findAllByPlatformId(platformId)
        val userCryptosIds = userCryptos.map { it.coingeckoCryptoId }.toTypedArray()
        val cryptos = cryptoService.findAllByIds(*userCryptosIds)
        val startValue = BigDecimal.ZERO

        val userCryptosQuantity = HashMap<String, BigDecimal>()
        userCryptos.forEach { userCrypto ->
            val coingeckoCryptoId = userCrypto.coingeckoCryptoId
            val cryptoQuantity = userCrypto.quantity

            // TODO - compare with java, there are differences
            userCryptosQuantity.compute(coingeckoCryptoId) { _, value -> startValue.add(value ?: cryptoQuantity) }
        }

        val totalBalances = getTotalBalances(cryptos, userCryptosQuantity)

        val cryptosInsights = cryptos.map { crypto ->
            val quantity = userCryptosQuantity[crypto.id] ?: BigDecimal.ZERO
            val cryptoTotalBalances = getCryptoTotalBalances(crypto, quantity)

            val cryptoInsight = CryptoInsights(
                cryptoName = crypto.name,
                cryptoId = crypto.id,
                quantity = quantity.toPlainString(),
                balances = cryptoTotalBalances,
                percentage = calculatePercentage(totalBalances.totalBalance, cryptoTotalBalances.totalBalance)
            )

            cryptoInsight
        }

        return PlatformInsightsResponse(
            platformName = platformResponse.name,
            balances = totalBalances,
            cryptos = cryptosInsights
        )
    }

    private fun getTotalBalances(cryptos: List<Crypto>, userCryptoQuantity: Map<String, BigDecimal>): Balances {
        var totalBalance = BigDecimal.ZERO
        var totalBTCBalance = BigDecimal.ZERO
        var totalEURBalance = BigDecimal.ZERO

        userCryptoQuantity.forEach { (coingeckoCryptoId, quantity) ->
            val crypto = cryptos.first { it.id.equals(coingeckoCryptoId, true) }
            val lastKnownPrice = crypto.lastKnownPrice
            val lastKnownPriceInBTC = crypto.lastKnownPriceInBTC
            val lastKnownPriceInEUR = crypto.lastKnownPriceInEUR

            totalBalance = totalBalance.add(lastKnownPrice.multiply(quantity))
            totalBTCBalance = totalBTCBalance.add(lastKnownPriceInBTC.multiply(quantity))
            totalEURBalance = totalEURBalance.add(lastKnownPriceInEUR.multiply(quantity))
        }

        return Balances(
            totalBalance = totalBalance.toPlainString(),
            totalBTCBalance = totalBTCBalance.toPlainString(),
            totalEURBalance = totalEURBalance.toPlainString()
        )
    }

    private fun getCryptoTotalBalances(crypto: Crypto, quantity: BigDecimal): Balances {
        return Balances(
            totalBalance = crypto.lastKnownPrice.multiply(quantity).toPlainString(),
            totalBTCBalance = crypto.lastKnownPriceInBTC.multiply(quantity).toPlainString(),
            totalEURBalance = crypto.lastKnownPriceInEUR.multiply(quantity).toPlainString()
        )
    }

    private fun calculatePercentage(totalBalance: String, cryptoBalance: String): Float {
        return BigDecimal(cryptoBalance)
            .multiply(BigDecimal("100"))
            .divide(BigDecimal(totalBalance), 2, RoundingMode.HALF_UP)
            .toFloat()
    }

}
